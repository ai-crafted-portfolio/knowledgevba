## 取り込み先

クラスモジュール（.cls）です。下記コードをコピーし、`clsKnowledgeManager.cls` というファイル名で保存して、VBE の「ファイル → ファイルのインポート」で取り込みます。先頭の `VERSION 1.0 CLASS` から始まる行はクラスモジュールのファイル形式の一部なので、削らずにそのまま保存してください。詳しい手順は[導入手順](../../setup.md)を参照してください。

VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
End
Attribute VB_Name = "clsKnowledgeManager"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = False
Attribute VB_Exposed = False
' ================================================================
' クラス: clsKnowledgeManager（v2.1、Phase 2 / 4、ナレッジ業務ロジック）
' 概要:   knowledge .txt の CRUD（v2.1 ###xxx### 新形式）+ 楽観ロック + 採番
' Version: v2.1（2026-05-16 EOD、Q1-Q57 解消反映）
' 関連:   modKnowledgeFileIO（Q47 新形式）, modFormatLoader（Q19）,
'         clsFieldMigrator（Q50 連動 migrate）
'         ADR-0053 §2.3 / §2.9 / §2.10
' v2.1 主要更新:
'   - Q47 knowledge .txt は ###xxx### 区切り新形式（modKnowledgeFileIO 経由）
'   - Q48 1 件 1 ファイル（data_dir 直下フラット）
'   - Q33 KnowledgeNo 採番 = <FormatID>-<連番 4 桁>
'   - Q45 DeleteKnowledge 追加（M-03 削除ボタン用）
'   - Q46 楽観ロック（GetKnowledgeTimestamp + SaveKnowledge originalTimestamp）
' ================================================================
Option Explicit

Private m_logger As Object  ' clsLogger 注入

' ----------------------------------------------------------------
' 初期化
' ----------------------------------------------------------------

Public Sub Init(Optional ByVal logger As Object = Nothing)
    Set m_logger = logger
End Sub

' ----------------------------------------------------------------
' Public API: 読込
' ----------------------------------------------------------------

' 概要: knowledge .txt を読込、Q46 楽観ロック用の timestamp を Out 引数で返す
' 引数: knowledgeNo / ByRef originalTimestamp（読込時の FileDateTime 保存用）
' 戻り値: Dictionary（Q47 新形式の key/value）
Public Function LoadKnowledge( _
    ByVal knowledgeNo As String, _
    Optional ByRef originalTimestamp As Date _
) As Object
    On Error GoTo ErrHandler
    Set LoadKnowledge = modKnowledgeFileIO.LoadKnowledge(knowledgeNo)
    originalTimestamp = modKnowledgeFileIO.GetKnowledgeTimestamp(knowledgeNo)
    Exit Function
ErrHandler:
    If Not m_logger Is Nothing Then
        m_logger.LogError "clsKnowledgeManager", "LoadKnowledge", Err.Description, "", "LOG-KNW-LOAD-ERR"
    End If
    Set LoadKnowledge = CreateObject("Scripting.Dictionary")
End Function

' ----------------------------------------------------------------
' Public API: 採番（Q33 確定、<FormatID>-<連番 4 桁>）
' ----------------------------------------------------------------

' 概要: 指定 FormatID で次の KnowledgeNo を採番
' 戻り値: "TICKET-0001" 形式、最大 9999 件 / FormatID
Public Function BuildKnowledgeNumber(ByVal formatId As String) As String
    On Error GoTo ErrHandler

    Dim existingList As Collection
    Set existingList = modKnowledgeFileIO.ListKnowledgesByFormat(formatId)

    Dim maxSeq As Long
    maxSeq = 0
    Dim knwNo As Variant
    For Each knwNo In existingList
        Dim seq As Long
        seq = ExtractSequence(CStr(knwNo), formatId)
        If seq > maxSeq Then maxSeq = seq
    Next knwNo

    Dim nextSeq As Long
    nextSeq = maxSeq + 1
    If nextSeq > 9999 Then
        Err.Raise vbObjectError + 200, "clsKnowledgeManager", _
                  "KnowledgeNo overflow: format " & formatId & " で 9999 件超過"
    End If

    BuildKnowledgeNumber = formatId & "-" & Format$(nextSeq, "0000")
    Exit Function

ErrHandler:
    If Not m_logger Is Nothing Then
        m_logger.LogError "clsKnowledgeManager", "BuildKnowledgeNumber", Err.Description, "", "LOG-KNW-BLDNO-ERR"
    End If
    BuildKnowledgeNumber = ""
End Function

Private Function ExtractSequence(ByVal knwNo As String, ByVal formatId As String) As Long
    On Error Resume Next
    Dim prefix As String
    prefix = formatId & "-"
    If Left(knwNo, Len(prefix)) <> prefix Then
        ExtractSequence = 0
        Exit Function
    End If
    Dim seqStr As String
    seqStr = Mid(knwNo, Len(prefix) + 1)
    ExtractSequence = CLng(seqStr)
    If Err.Number <> 0 Then
        ExtractSequence = 0
        Err.Clear
    End If
End Function

' ----------------------------------------------------------------
' Public API: 保存（Q46 楽観ロック対応）
' ----------------------------------------------------------------

' 概要: knowledge を保存、楽観ロック check で競合検出
' 引数: knowledgeNo / knowledgeDict / originalTimestamp（読込時の値、0 = 新規）
' 戻り値: 0=OK / 1=競合検出 / 2=その他エラー
Public Function SaveKnowledge( _
    ByVal knowledgeNo As String, _
    ByVal knowledgeDict As Object, _
    ByVal originalTimestamp As Date _
) As Long
    On Error GoTo ErrHandler

    ' format 保存時 migrate 連動（Q50）は呼出側 clsFormatManager の責務、
    ' 本 method は素朴に modKnowledgeFileIO.SaveKnowledge を呼ぶ

    SaveKnowledge = modKnowledgeFileIO.SaveKnowledge(knowledgeNo, knowledgeDict, originalTimestamp)

    If SaveKnowledge = 0 And Not m_logger Is Nothing Then
        m_logger.LogInfo "clsKnowledgeManager", "SaveKnowledge", "保存: " & knowledgeNo, knowledgeNo, "LOG-KNW-SAVE-OK"
    ElseIf SaveKnowledge = 1 And Not m_logger Is Nothing Then
        m_logger.LogWarn "clsKnowledgeManager", "SaveKnowledge", "競合検出: " & knowledgeNo, knowledgeNo, "LOG-KNW-CONFLICT"
    End If
    Exit Function

ErrHandler:
    If Not m_logger Is Nothing Then
        m_logger.LogError "clsKnowledgeManager", "SaveKnowledge", Err.Description, "", "LOG-KNW-SAVE-ERR"
    End If
    SaveKnowledge = 2
End Function

' ----------------------------------------------------------------
' Public API: 削除（Q45 M-03 削除ボタン用）
' ----------------------------------------------------------------

' 概要: knowledge 削除（削除前に自動バックアップ取得）
' 引数: knowledgeNo
' 戻り値: True = 成功
Public Function DeleteKnowledge(ByVal knowledgeNo As String) As Boolean
    On Error GoTo ErrHandler

    ' Q45 削除前自動バックアップ
    Dim ts As String
    ts = Format$(Now(), "yyyymmdd_hhnnss")
    If Not modKnowledgeFileIO.BackupKnowledgeFile(knowledgeNo, ts) Then
        If Not m_logger Is Nothing Then
            m_logger.LogError "clsKnowledgeManager", "DeleteKnowledge", _
                              "削除前バックアップ失敗: " & knowledgeNo, knowledgeNo, "LOG-KNW-DEL-NOTFOUND"
        End If
        DeleteKnowledge = False
        Exit Function
    End If

    ' 実削除 (modKnowledgeFileIO returns Long, 0=success)
    DeleteKnowledge = (modKnowledgeFileIO.DeleteKnowledge(knowledgeNo) = 0)

    If DeleteKnowledge And Not m_logger Is Nothing Then
        m_logger.LogInfo "clsKnowledgeManager", "DeleteKnowledge", _
                         "削除: " & knowledgeNo & " (backup_ts=" & ts & ")", _
                         knowledgeNo, "LOG-KNW-DEL-OK"
    End If
    Exit Function

ErrHandler:
    If Not m_logger Is Nothing Then
        m_logger.LogError "clsKnowledgeManager", "DeleteKnowledge", Err.Description, "", "LOG-KNW-DEL-ERR"
    End If
    DeleteKnowledge = False
End Function

' ----------------------------------------------------------------
' Public API: 一覧取得
' ----------------------------------------------------------------

Public Function ListAllKnowledges() As Collection
    Set ListAllKnowledges = modKnowledgeFileIO.ListAllKnowledges()
End Function

Public Function ListKnowledgesByFormat(ByVal formatId As String) As Collection
    Set ListKnowledgesByFormat = modKnowledgeFileIO.ListKnowledgesByFormat(formatId)
End Function
