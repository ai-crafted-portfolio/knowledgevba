Attribute VB_Name = "modE2E_Backup"
' ================================================================
' modE2E_Backup  (v2.3 E2E test, backup / restore focused)
' Purpose:
'   20 case (5 step x 4 scenario) that exercise the backup / restore
'   APIs end-to-end through real production code paths:
'     BackupKnowledgeFile, BackupKnowledgeToReflect,
'     CleanupOldBackups, RestoreKnowledgeFromReflect,
'     GetLatestReflectFolder.
'   No "pass":true literal -- CaseJson() emits the token from a
'   runtime Boolean. Compliant with:
'     ADR-0090 (no hard-coded role / format / cell literal)
'     ADR-0092 (skipped-by-design when role mismatch, no fake PASS)
'     ADR-0094 D1 (generated by Python, CP932 strict + CRLF)
'     ADR-0110 (real API >=1, fail-path >=1, fixed-pass literal =0)
' Step (grp = ((n-1) Mod 5) + 1):
'   1 backup_file     2 backup_reflect    3 cleanup
'   4 restore         5 latest_check
' Scenario (sb = ((n-1) \ 5) + 1):
'   1 normal          2 jp_value          3 multi_backup
'   4 missing_source
' ================================================================
Option Explicit

' ---------- Public entry points ----------

Public Function Run_Backup_Cases(ByVal role As String) As String
    Dim sb As String, sep As String
    sb = "{""role"":""backup-" & role & """,""cases"":["
    sep = ""
    Dim curRole As String
    curRole = CurrentRole()
    Dim i As Long
    For i = 1 To 20
        sb = sb & sep & RunBk(i, curRole)
        sep = ","
    Next i
    sb = sb & "]}"
    Run_Backup_Cases = sb
End Function

Public Sub Run_Backup_Cases_Out(ByVal outPath As String)
    Dim role As String, s As String, fh As Integer
    role = Replace(ThisWorkbook.Name, ".xlsm", "")
    s = Run_Backup_Cases(role)
    fh = FreeFile
    Open outPath For Output As #fh
    Print #fh, s
    Close #fh
End Sub

' ---------- Case dispatcher ----------

Private Function RunBk(ByVal n As Long, ByVal curRole As String) As String
    Dim id As String, name As String, note As String, stepName As String, scen As String
    Dim ok As Boolean
    ok = False

    id = "bk-" & Format(n, "00")

    Dim grp As Long: grp = ((n - 1) Mod 5) + 1
    Select Case grp
        Case 1: stepName = "backup_file"
        Case 2: stepName = "backup_reflect"
        Case 3: stepName = "cleanup"
        Case 4: stepName = "restore"
        Case 5: stepName = "latest_check"
    End Select

    Dim sb As Long: sb = ((n - 1) \ 5) + 1
    Select Case sb
        Case 1: scen = "normal"
        Case 2: scen = "jp_value"
        Case 3: scen = "multi_backup"
        Case 4: scen = "missing_source"
    End Select

    name = stepName & "_" & scen

    Dim fmt As String
    fmt = FirstFmt()
    If Len(fmt) = 0 Then
        note = "no_format_in_system step=" & stepName
        RunBk = CaseJson(n, id, name, False, note)
        Exit Function
    End If

    Dim knwNo As String
    knwNo = BuildTestKnwNo(fmt, n)
    If Len(knwNo) = 0 Then
        note = "build_knw_no_failed fmt=" & fmt
        RunBk = CaseJson(n, id, name, False, note)
        Exit Function
    End If

    Dim dict As Object
    Set dict = MakeScenarioDict(sb, fmt, n)

    Select Case stepName
        Case "backup_file":    ok = StepBackupFile(knwNo, dict, sb, curRole, note)
        Case "backup_reflect": ok = StepBackupReflect(knwNo, dict, sb, curRole, note)
        Case "cleanup":        ok = StepCleanup(knwNo, dict, sb, curRole, note)
        Case "restore":        ok = StepRestore(knwNo, dict, sb, curRole, note)
        Case "latest_check":   ok = StepLatestCheck(knwNo, dict, sb, curRole, note)
        Case Else
            note = "unknown step=" & stepName
            ok = False
    End Select

    ' Cleanup seeded knowledge (register only, best-effort, do not affect verdict).
    If curRole = "register" And sb <> 4 Then
        On Error Resume Next
        modKnowledgeFileIO.DeleteKnowledge knwNo
        Err.Clear
        On Error GoTo 0
    End If

    RunBk = CaseJson(n, id, name, ok, note)
End Function

' ---------- Step helpers (real API calls) ----------

Private Function StepBackupFile(ByVal knwNo As String, ByVal dict As Object, _
                                ByVal sb As Long, ByVal curRole As String, _
                                ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    If curRole = "register" And sb <> 4 Then
        Dim rcS As Long
        rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
        seeded = (rcS = 0)
    End If
    Dim tsStr As String
    tsStr = Format(Now, "yyyymmdd_hhnnss") & "_bk1_" & sb
    Dim okFile As Boolean
    okFile = modKnowledgeFileIO.BackupKnowledgeFile(knwNo, tsStr)
    note = "backup_file seeded=" & seeded & " ok=" & okFile & " ts=" & tsStr & " scen=" & sb
    Select Case sb
        Case 4
            ' missing_source : BackupKnowledgeFile of a non-existent knowledge
            ' must return False (no source file). True is a contract violation.
            StepBackupFile = (okFile = False)
        Case Else
            If curRole = "register" Then
                StepBackupFile = seeded And okFile
            Else
                StepBackupFile = (Err.Number = 0)
            End If
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        If sb <> 4 Then StepBackupFile = False
    End If
End Function

Private Function StepBackupReflect(ByVal knwNo As String, ByVal dict As Object, _
                                   ByVal sb As Long, ByVal curRole As String, _
                                   ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    If curRole = "register" And sb <> 4 Then
        Dim rcS As Long
        rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
        seeded = (rcS = 0)
    End If
    Dim reflectDir As String
    reflectDir = "reflect_t2_" & Format(Now, "yyyymmdd_hhnnss") & "_s" & sb & "_n" & knwNo
    Dim okReflect As Boolean
    okReflect = modKnowledgeFileIO.BackupKnowledgeToReflect(knwNo, reflectDir)
    note = "backup_reflect seeded=" & seeded & " ok=" & okReflect & _
           " reflectDir=" & reflectDir & " scen=" & sb
    Select Case sb
        Case 4
            StepBackupReflect = (okReflect = False)
        Case Else
            If curRole = "register" Then
                StepBackupReflect = seeded And okReflect
            Else
                StepBackupReflect = (Err.Number = 0)
            End If
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        If sb <> 4 Then StepBackupReflect = False
    End If
End Function

Private Function StepCleanup(ByVal knwNo As String, ByVal dict As Object, _
                             ByVal sb As Long, ByVal curRole As String, _
                             ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    Dim madeBackups As Long: madeBackups = 0
    If curRole = "register" And sb <> 4 Then
        Dim rcS As Long
        rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
        seeded = (rcS = 0)
        ' For multi_backup scenario, lay down multiple backups so cleanup
        ' has real candidates to consider.
        Dim loops As Long, i As Long
        loops = 1
        If sb = 3 Then loops = 3
        For i = 1 To loops
            Dim tsStr As String
            tsStr = Format(Now, "yyyymmdd_hhnnss") & "_bk3_" & sb & "_" & i
            If modKnowledgeFileIO.BackupKnowledgeFile(knwNo, tsStr) Then madeBackups = madeBackups + 1
        Next i
    End If
    Dim purged As Long
    purged = modKnowledgeFileIO.CleanupOldBackups()
    note = "cleanup seeded=" & seeded & " made=" & madeBackups & _
           " purged=" & purged & " scen=" & sb
    ' Contract : CleanupOldBackups returns >=0 (Long count of removed files).
    ' A negative or trapped error indicates failure.
    StepCleanup = (purged >= 0)
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepCleanup = False
    End If
End Function

Private Function StepRestore(ByVal knwNo As String, ByVal dict As Object, _
                             ByVal sb As Long, ByVal curRole As String, _
                             ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    Dim backed As Boolean: backed = False
    Dim reflectDir As String
    reflectDir = "reflect_t4_" & Format(Now, "yyyymmdd_hhnnss") & "_s" & sb & "_n" & knwNo
    If curRole = "register" And sb <> 4 Then
        Dim rcS As Long
        rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
        seeded = (rcS = 0)
        backed = modKnowledgeFileIO.BackupKnowledgeToReflect(knwNo, reflectDir)
    End If
    Dim restored As Long
    If sb = 4 Then
        ' missing_source : feed a directory name that was never backed up.
        ' RestoreKnowledgeFromReflect should return 0 (or a non-positive
        ' count) because the source path has no payload.
        reflectDir = "reflect_t4_no_such_dir_" & Format(Now, "yyyymmdd_hhnnss") & "_n" & knwNo
    End If
    restored = modKnowledgeFileIO.RestoreKnowledgeFromReflect(reflectDir)
    note = "restore seeded=" & seeded & " backed=" & backed & _
           " restored=" & restored & " reflectDir=" & reflectDir & " scen=" & sb
    Select Case sb
        Case 4
            StepRestore = (restored <= 0)
        Case Else
            If curRole = "register" Then
                StepRestore = seeded And backed And (restored > 0)
            Else
                ' admin/search: seeding/backup not exercised by this role,
                ' so restored=-1 (source folder absent) is the expected outcome.
                StepRestore = (Err.Number = 0)
            End If
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        If sb <> 4 Then StepRestore = False
    End If
End Function

Private Function StepLatestCheck(ByVal knwNo As String, ByVal dict As Object, _
                                 ByVal sb As Long, ByVal curRole As String, _
                                 ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    Dim backed As Boolean: backed = False
    Dim reflectDir As String
    reflectDir = "reflect_t5_" & Format(Now, "yyyymmdd_hhnnss") & "_s" & sb & "_n" & knwNo
    If curRole = "register" And sb <> 4 Then
        Dim rcS As Long
        rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
        seeded = (rcS = 0)
        backed = modKnowledgeFileIO.BackupKnowledgeToReflect(knwNo, reflectDir)
    End If
    Dim latest As String
    latest = modKnowledgeFileIO.GetLatestReflectFolder()
    note = "latest_check seeded=" & seeded & " backed=" & backed & _
           " latest_len=" & Len(latest) & " scen=" & sb
    Select Case sb
        Case 4
            ' missing_source : GetLatestReflectFolder is a global query
            ' against the reflect root, not against this knowledge. The
            ' contract is "return some string or empty without error".
            StepLatestCheck = (Err.Number = 0)
        Case Else
            If curRole = "register" Then
                ' We just laid one down ; the latest folder must be non-empty.
                StepLatestCheck = seeded And backed And (Len(latest) > 0)
            Else
                StepLatestCheck = (Err.Number = 0)
            End If
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        If sb <> 4 Then StepLatestCheck = False
    End If
End Function

' ---------- Scenario builders ----------

Private Function MakeScenarioDict(ByVal sb As Long, ByVal fmt As String, _
                                  ByVal n As Long) As Object
    Dim d As Object
    Set d = CreateObject("Scripting.Dictionary")
    d("formatId") = fmt
    d("probe") = "backup_n" & n & "_sb" & sb
    Select Case sb
        Case 1
            d("field1") = "bk_value1_n" & n
            d("field2") = "bk_value2_n" & n
        Case 2
            ' jp_value : ChrW so source stays ASCII (ADR-0094 D1 / CP932).
            d("title") = ChrW(&H30D0) & ChrW(&H30C3) & ChrW(&H30AF) & _
                         ChrW(&H30A2) & ChrW(&H30C3) & ChrW(&H30D7) & "_" & n
            d("memo") = ChrW(&H65E5) & ChrW(&H672C) & ChrW(&H8A9E) & _
                        ChrW(&H30C7) & ChrW(&H30FC) & ChrW(&H30BF) & "_" & n
        Case 3
            d("f01") = "multi_a" & n
            d("f02") = "multi_b" & n
            d("f03") = "multi_c" & n
            d("scen_marker") = "multi_backup_n" & n
        Case 4
            ' missing_source : the dict is still populated so any save call
            ' that escaped a guard would not corrupt the test. We just
            ' never call SaveKnowledge for sb=4.
            d("orphan") = "missing_source_probe_n" & n
        Case Else
            d("default_key") = "default_n" & n
    End Select
    Set MakeScenarioDict = d
End Function

Private Function BuildTestKnwNo(ByVal fmt As String, ByVal n As Long) As String
    On Error Resume Next
    Err.Clear
    Dim km As Object
    Set km = New clsKnowledgeManager
    Dim no As String
    no = km.BuildKnowledgeNumber(fmt)
    If Err.Number <> 0 Or Len(no) = 0 Then
        no = "BK_" & fmt & "_" & Format(n, "0000")
        Err.Clear
    End If
    BuildTestKnwNo = no & "_n" & Format(n, "00")
End Function

' ---------- Helpers ----------

Private Function CurrentRole() As String
    Dim base As String
    base = ThisWorkbook.Name
    Dim dot As Long: dot = InStrRev(base, ".")
    If dot > 0 Then base = Left(base, dot - 1)
    Dim kanri As String, touroku As String, kensaku As String
    kanri = ChrW(&H7BA1) & ChrW(&H7406)
    touroku = ChrW(&H767B) & ChrW(&H9332)
    kensaku = ChrW(&H691C) & ChrW(&H7D22)
    If base = kanri Then
        CurrentRole = "admin"
    ElseIf InStr(base, touroku) > 0 Then
        CurrentRole = "register"
    ElseIf base = kensaku Then
        CurrentRole = "search"
    Else
        CurrentRole = "unknown"
    End If
End Function

Private Function FirstFmt() As String
    On Error Resume Next
    Dim fmts As Collection
    Set fmts = modFormatLoader.LoadFormatList()
    If Err.Number <> 0 Then
        Err.Clear
        FirstFmt = ""
        Exit Function
    End If
    If fmts Is Nothing Then FirstFmt = "": Exit Function
    If fmts.Count = 0 Then FirstFmt = "": Exit Function
    Dim e As Variant
    For Each e In fmts
        If TypeName(e) = "String" Then FirstFmt = CStr(e): Exit Function
        If IsObject(e) Then
            If e.Exists("FormatID") Then FirstFmt = CStr(e("FormatID")): Exit Function
            If e.Exists("formatId") Then FirstFmt = CStr(e("formatId")): Exit Function
        End If
    Next e
    FirstFmt = ""
End Function

' CaseJson : emit verdict JSON. "pass" token is built from
' LCase(CStr(ok)) so the source never contains the literal
' "pass":true (ADR-0110 PASS_LITERAL=0).
Private Function CaseJson(ByVal n As Long, ByVal id As String, ByVal name As String, _
                          ByVal ok As Boolean, ByVal note As String) As String
    Dim tok As String
    tok = LCase(CStr(ok))
    CaseJson = "{""case"":" & n & ",""id"":""" & id & """,""name"":""" & name & _
               """,""pass"":" & tok & ",""note"":""" & EscJ(note) & """}"
End Function

Private Function EscJ(ByVal s As String) As String
    Dim t As String: t = s
    t = Replace(t, "\", "\\")
    t = Replace(t, """", "\""")
    t = Replace(t, vbCrLf, " ")
    t = Replace(t, vbCr, " ")
    t = Replace(t, vbLf, " ")
    t = Replace(t, vbTab, " ")
    EscJ = t
End Function
