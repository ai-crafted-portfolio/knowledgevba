## ソースコード

```vb
Attribute VB_Name = "modE2E_Lifecycle"
' ================================================================
' modE2E_Lifecycle  (v2.3 E2E test, knowledge full lifecycle)
' Purpose:
'   30 case (6 step x 5 scenario) that drive a knowledge record
'   create -> read -> update -> backup -> restore -> delete via the
'   real production APIs (modKnowledgeFileIO, modFormatLoader,
'   clsKnowledgeManager). No "pass":true literal -- the helper
'   CaseJson() converts a runtime Boolean into the JSON token at
'   emit time. Compliant with:
'     ADR-0090 (no hard-coded role / format / cell literal)
'     ADR-0092 (no fake PASS, real negative assertions)
'     ADR-0094 D1 (generated by Python, CP932 strict + CRLF)
'     ADR-0110 (real API >=1, fail-path >=1, fixed-pass literal =0)
' Role gate behaviour (strengthened):
'   register : runs the full 6-step lifecycle on all 30 cases.
'   admin/search : runs negative / structural probes against the
'                  real APIs (no skipped-by-design weak passes):
'                    create/update/delete -> gate must reject (rc=3
'                                            / DeleteKnowledge=False)
'                    read                  -> LoadKnowledge dict
'                                            Count > 0 AND formatId
'                                            key exists
'                    backup                -> BackupKnowledgeFile +
'                                            FSO.FileExists on the
'                                            produced .bak path
'                    restore               -> RestoreFromReflect count
'                                            parity vs reflect dir
' ================================================================
Option Explicit

' ---------- Public entry points ----------

Public Function Run_Lifecycle_Cases(ByVal role As String) As String
    Dim sb As String, sep As String
    sb = "{""role"":""lifecycle-" & role & """,""cases"":["
    sep = ""
    Dim curRole As String
    curRole = CurrentRole()
    Dim i As Long
    For i = 1 To 30
        sb = sb & sep & RunLc(i, curRole)
        sep = ","
    Next i
    sb = sb & "]}"
    Run_Lifecycle_Cases = sb
End Function

Public Sub Run_Lifecycle_Cases_Out(ByVal outPath As String)
    Dim role As String, s As String, fh As Integer
    role = Replace(ThisWorkbook.Name, ".xlsm", "")
    s = Run_Lifecycle_Cases(role)
    fh = FreeFile
    Open outPath For Output As #fh
    Print #fh, s
    Close #fh
End Sub

' ---------- Case dispatcher ----------

Private Function RunLc(ByVal n As Long, ByVal curRole As String) As String
    Dim id As String, name As String, note As String, stepName As String, scen As String
    Dim ok As Boolean
    ' Default ok = False ; each step helper must set True on success.
    ' Unhandled paths therefore emit FAIL JSON.
    ok = False

    id = "lc-" & Format(n, "00")

    Dim grp As Long: grp = ((n - 1) Mod 6) + 1
    Select Case grp
        Case 1: stepName = "create"
        Case 2: stepName = "read"
        Case 3: stepName = "update"
        Case 4: stepName = "backup"
        Case 5: stepName = "restore"
        Case 6: stepName = "delete"
    End Select

    Dim sb As Long: sb = ((n - 1) \ 6) + 1
    Select Case sb
        Case 1: scen = "normal"
        Case 2: scen = "jp_unicode"
        Case 3: scen = "boundary_max"
        Case 4: scen = "multiple_field"
        Case 5: scen = "round_trip_strict"
    End Select

    name = stepName & "_" & scen

    Dim fmt As String
    fmt = FirstFmt()
    If Len(fmt) = 0 Then
        note = "no_format_in_system step=" & stepName
        RunLc = CaseJson(n, id, name, False, note)
        Exit Function
    End If

    Dim knwNo As String
    knwNo = BuildScenarioKnwNo(fmt, n, sb)
    If Len(knwNo) = 0 Then
        note = "build_knw_no_failed fmt=" & fmt
        RunLc = CaseJson(n, id, name, False, note)
        Exit Function
    End If

    Dim dict As Object
    Set dict = MakeScenarioDict(sb, fmt, n)

    ' --------------------------------------------------------------
    ' Role-aware dispatch.
    ' register : positive lifecycle (existing helpers).
    ' admin/search : strengthened negative / structural probes that
    '                exercise the real APIs and assert specific
    '                return-code / dict / file-system contracts.
    ' --------------------------------------------------------------
    If curRole = "register" Then
        Select Case stepName
            Case "create":  ok = StepCreate(knwNo, dict, sb, note)
            Case "read":    ok = StepRead(knwNo, dict, sb, note)
            Case "update":  ok = StepUpdate(knwNo, dict, sb, note)
            Case "backup":  ok = StepBackup(knwNo, dict, sb, note)
            Case "restore": ok = StepRestore(knwNo, dict, sb, note)
            Case "delete":  ok = StepDelete(knwNo, dict, sb, note)
            Case Else
                note = "unknown step=" & stepName
                ok = False
        End Select
    Else
        ' admin / search / unknown -- all funnel through the same
        ' strengthened probe set. The probes either expect gate
        ' rejection or structural validity ; they never short-circuit
        ' on Err.Number=0.
        Select Case stepName
            Case "create":  ok = ProbeNonRegister_Create(knwNo, dict, sb, note)
            Case "read":    ok = ProbeNonRegister_Read(knwNo, dict, sb, note)
            Case "update":  ok = ProbeNonRegister_Update(knwNo, dict, sb, note)
            Case "backup":  ok = ProbeNonRegister_Backup(knwNo, dict, sb, note)
            Case "restore": ok = ProbeNonRegister_Restore(knwNo, dict, sb, note)
            Case "delete":  ok = ProbeNonRegister_Delete(knwNo, dict, sb, note)
            Case Else
                note = "unknown step=" & stepName
                ok = False
        End Select
    End If

    ' Cleanup : best-effort delete (only on register role). Tolerate
    ' role gate / missing file -- do not affect the verdict.
    If curRole = "register" Then
        On Error Resume Next
        modKnowledgeFileIO.DeleteKnowledge knwNo
        Err.Clear
        On Error GoTo 0
    End If

    RunLc = CaseJson(n, id, name, ok, note)
End Function

' ---------- Step helpers (register role : positive lifecycle) ----------

Private Function StepCreate(ByVal knwNo As String, ByVal dict As Object, _
                            ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim rc As Long
    rc = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim loaded As Object
    Set loaded = modKnowledgeFileIO.LoadKnowledge(knwNo)
    Dim hasIt As Boolean
    hasIt = (Not loaded Is Nothing)
    If hasIt Then
        If TypeName(loaded) = "Dictionary" Then
            hasIt = (loaded.Count > 0)
        End If
    End If
    note = "create rc=" & rc & " scen=" & sb & " loaded_present=" & hasIt
    StepCreate = (rc = 0) And hasIt
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepCreate = False
    End If
End Function

Private Function StepRead(ByVal knwNo As String, ByVal dict As Object, _
                          ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seeded As Boolean: seeded = False
    Dim rcSeed As Long
    rcSeed = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    seeded = (rcSeed = 0)
    Dim loaded As Object
    Set loaded = modKnowledgeFileIO.LoadKnowledge(knwNo)
    Dim cnt As Long: cnt = 0
    If Not loaded Is Nothing Then
        If TypeName(loaded) = "Dictionary" Then cnt = loaded.Count
    End If
    Dim hasKey As Boolean
    hasKey = False
    If cnt > 0 Then
        Dim k As Variant
        For Each k In dict.Keys
            If loaded.Exists(k) Then
                hasKey = True
                Exit For
            End If
        Next k
    End If
    note = "read seeded=" & seeded & " loaded_cnt=" & cnt & " key_overlap=" & hasKey
    StepRead = seeded And (cnt > 0) And hasKey
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepRead = False
    End If
End Function

Private Function StepUpdate(ByVal knwNo As String, ByVal dict As Object, _
                            ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim rc1 As Long, rc2 As Long
    rc1 = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim ts As Date
    ts = modKnowledgeFileIO.GetKnowledgeTimestamp(knwNo)
    dict("scen_marker") = "updated_" & sb
    rc2 = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, ts)
    Dim loaded As Object
    Set loaded = modKnowledgeFileIO.LoadKnowledge(knwNo)
    Dim observedMarker As String
    observedMarker = ""
    If Not loaded Is Nothing Then
        If loaded.Exists("scen_marker") Then observedMarker = CStr(loaded("scen_marker"))
    End If
    note = "update rc1=" & rc1 & " rc2=" & rc2 & " marker=" & observedMarker
    StepUpdate = (rc1 = 0) And (rc2 = 0) And (observedMarker = "updated_" & sb)
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepUpdate = False
    End If
End Function

Private Function StepBackup(ByVal knwNo As String, ByVal dict As Object, _
                            ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim rcS As Long
    rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim seeded As Boolean
    seeded = (rcS = 0)
    Dim okFile As Boolean, okReflect As Boolean
    Dim tsStr As String
    tsStr = Format(Now, "yyyymmdd_hhnnss") & "_lc" & sb
    okFile = modKnowledgeFileIO.BackupKnowledgeFile(knwNo, tsStr)
    Dim reflectDir As String
    reflectDir = "lc_" & Format(Now, "yyyymmdd_hhnnss") & "_" & sb
    okReflect = modKnowledgeFileIO.BackupKnowledgeToReflect(knwNo, reflectDir)
    note = "backup seeded=" & seeded & " file=" & okFile & " reflect=" & okReflect & _
           " reflectDir=" & reflectDir
    StepBackup = seeded And okFile And okReflect
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepBackup = False
    End If
End Function

Private Function StepRestore(ByVal knwNo As String, ByVal dict As Object, _
                             ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim reflectDir As String
    reflectDir = "lcr_" & Format(Now, "yyyymmdd_hhnnss") & "_" & sb
    Dim rcS As Long
    rcS = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim seeded As Boolean
    seeded = (rcS = 0)
    Dim backed As Boolean
    backed = modKnowledgeFileIO.BackupKnowledgeToReflect(knwNo, reflectDir)
    Dim restored As Long
    restored = modKnowledgeFileIO.RestoreKnowledgeFromReflect(reflectDir)
    note = "restore seeded=" & seeded & " backed=" & backed & " restored_cnt=" & restored & _
           " reflectDir=" & reflectDir
    StepRestore = seeded And backed And (restored > 0)
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepRestore = False
    End If
End Function

Private Function StepDelete(ByVal knwNo As String, ByVal dict As Object, _
                            ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim rc As Long
    rc = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim okDel As Boolean
    okDel = modKnowledgeFileIO.DeleteKnowledge(knwNo)
    Dim postLoad As Object
    Set postLoad = modKnowledgeFileIO.LoadKnowledge(knwNo)
    Dim absent As Boolean
    absent = False
    If postLoad Is Nothing Then
        absent = True
    Else
        If TypeName(postLoad) = "Dictionary" Then
            absent = (postLoad.Count = 0)
        End If
    End If
    note = "delete seed_rc=" & rc & " del=" & okDel & " absent_after=" & absent
    StepDelete = (rc = 0) And okDel And absent
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        StepDelete = False
    End If
End Function

' ---------- Probe helpers (admin / search role : negative / structural) ----------

' create on non-register role : SaveKnowledge must be gate-rejected
' (rc=3). rc=0 means the role gate leaked -> FAIL.
Private Function ProbeNonRegister_Create(ByVal knwNo As String, ByVal dict As Object, _
                                         ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim rc As Long
    rc = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim ok As Boolean
    If rc = 3 Then
        ok = True
        note = "gate_rejected rc=3 scen=" & sb
    ElseIf rc = 0 Then
        ok = False
        note = "gate_LEAK rc=0 (accepted on non-register) scen=" & sb
    Else
        ok = False
        note = "gate_unexpected_rc=" & rc & " scen=" & sb
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Create = ok
End Function

' update on non-register role : SaveKnowledge with same id must be
' gate-rejected (rc=3). rc=0 means the gate leaked -> FAIL.
Private Function ProbeNonRegister_Update(ByVal knwNo As String, ByVal dict As Object, _
                                         ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    dict("scen_marker") = "updated_" & sb
    Dim rc As Long
    rc = modKnowledgeFileIO.SaveKnowledge(knwNo, dict, 0)
    Dim ok As Boolean
    If rc = 3 Then
        ok = True
        note = "gate_rejected rc=3 (update) scen=" & sb
    ElseIf rc = 0 Then
        ok = False
        note = "gate_LEAK rc=0 (update accepted) scen=" & sb
    Else
        ok = False
        note = "gate_unexpected_rc=" & rc & " (update) scen=" & sb
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Update = ok
End Function

' delete on non-register role : DeleteKnowledge must return False
' (gate-rejected). True on non-register means the gate leaked -> FAIL.
Private Function ProbeNonRegister_Delete(ByVal knwNo As String, ByVal dict As Object, _
                                         ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim okDel As Boolean
    okDel = modKnowledgeFileIO.DeleteKnowledge(knwNo)
    Dim ok As Boolean
    If okDel = False Then
        ok = True
        note = "gate_rejected del=False scen=" & sb
    Else
        ok = False
        note = "gate_LEAK del=True (non-register accepted) scen=" & sb
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Delete = ok
End Function

' read on non-register role : LoadKnowledge must return a Dictionary
' object whose Count > 0 AND that contains the "formatId" key. If the
' record is absent (admin/search xlsm has its own storage) the probe
' falls back to the cross-role contract : an empty Dictionary -> FAIL.
Private Function ProbeNonRegister_Read(ByVal knwNo As String, ByVal dict As Object, _
                                       ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim seedNo As String
    seedNo = FirstExistingKnwNo()
    Dim targetNo As String
    If Len(seedNo) > 0 Then
        targetNo = seedNo
    Else
        targetNo = knwNo
    End If
    Dim got As Object
    Set got = modKnowledgeFileIO.LoadKnowledge(targetNo)
    Dim cnt As Long: cnt = 0
    Dim hasFmtKey As Boolean: hasFmtKey = False
    If Not got Is Nothing Then
        If TypeName(got) = "Dictionary" Then
            cnt = got.Count
            If cnt > 0 Then
                hasFmtKey = got.Exists("formatId")
            End If
        End If
    End If
    Dim ok As Boolean
    If cnt > 0 And hasFmtKey Then
        ok = True
        note = "load_dict_valid count=" & cnt & " has_formatId=True scen=" & sb & " target=" & targetNo
    ElseIf cnt > 0 And Not hasFmtKey Then
        ok = False
        note = "load_dict_missing_formatId count=" & cnt & " scen=" & sb & " target=" & targetNo
    ElseIf got Is Nothing Then
        ok = False
        note = "load_returned_nothing scen=" & sb & " target=" & targetNo
    Else
        ok = False
        note = "load_dict_empty count=0 scen=" & sb & " target=" & targetNo
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Read = ok
End Function

' backup on non-register role : BackupKnowledgeFile against an
' existing knwNo must return True AND produce a file on disk that
' FSO.FileExists confirms. A non-existent knwNo must return False.
Private Function ProbeNonRegister_Backup(ByVal knwNo As String, ByVal dict As Object, _
                                         ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim existing As String
    existing = FirstExistingKnwNo()
    Dim tsStr As String
    tsStr = Format(Now, "yyyymmdd_hhnnss") & "_lcp" & sb
    Dim ok As Boolean
    Dim okFile As Boolean
    Dim fsoFound As Boolean: fsoFound = False
    If Len(existing) > 0 Then
        okFile = modKnowledgeFileIO.BackupKnowledgeFile(existing, tsStr)
        If okFile Then
            fsoFound = BackupFileFsoExists(existing, tsStr)
            ok = fsoFound
            If ok Then
                note = "backup_existing ok=True fso_exists=True scen=" & sb & " target=" & existing
            Else
                note = "backup_existing ok=True fso_exists=False (file missing) scen=" & sb & " target=" & existing
            End If
        Else
            ok = False
            note = "backup_existing ok=False (expected True) scen=" & sb & " target=" & existing
        End If
    Else
        ' Fall back to non-existent knwNo : the API must return False.
        Dim fakeNo As String
        fakeNo = "LC_NONEXIST_" & Format(Now, "yyyymmddhhnnss") & "_" & sb
        okFile = modKnowledgeFileIO.BackupKnowledgeFile(fakeNo, tsStr)
        If okFile = False Then
            ok = True
            note = "backup_nonexistent ok=False (as expected) scen=" & sb & " target=" & fakeNo
        Else
            ok = False
            note = "backup_nonexistent ok=True (UNEXPECTED -- no source file) scen=" & sb & " target=" & fakeNo
        End If
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Backup = ok
End Function

' restore on non-register role : prepare a reflect dir, count its
' members via FSO, then call RestoreKnowledgeFromReflect ; restored
' count must be >= the dir file count (parity assertion). If the
' reflect dir is empty / missing then restored must be 0.
Private Function ProbeNonRegister_Restore(ByVal knwNo As String, ByVal dict As Object, _
                                          ByVal sb As Long, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim reflectDir As String
    reflectDir = "lcp_" & Format(Now, "yyyymmdd_hhnnss") & "_" & sb
    Dim existing As String
    existing = FirstExistingKnwNo()
    Dim staged As Boolean: staged = False
    Dim expectedFiles As Long: expectedFiles = 0
    If Len(existing) > 0 Then
        ' BackupKnowledgeToReflect is gate-allowed for non-mutating
        ' use ; if the runtime gates it on non-register the probe
        ' degrades to the empty-dir branch (expectedFiles=0).
        staged = modKnowledgeFileIO.BackupKnowledgeToReflect(existing, reflectDir)
        If staged Then
            expectedFiles = ReflectDirFileCount(reflectDir)
        End If
    End If
    Dim restored As Long
    restored = modKnowledgeFileIO.RestoreKnowledgeFromReflect(reflectDir)
    Dim ok As Boolean
    If expectedFiles > 0 Then
        ok = (restored >= expectedFiles)
        note = "restore_parity expected=" & expectedFiles & " restored=" & restored & _
               " scen=" & sb & " reflectDir=" & reflectDir
    Else
        ' No files staged : a well-behaved API must report 0 restored.
        If restored = 0 Then
            ok = True
            note = "restore_empty expected=0 restored=0 scen=" & sb & " reflectDir=" & reflectDir
        Else
            ok = False
            note = "restore_empty UNEXPECTED restored=" & restored & " scen=" & sb & " reflectDir=" & reflectDir
        End If
    End If
    If Err.Number <> 0 Then
        ok = False
        note = note & " err=" & Err.Description
        Err.Clear
    End If
    ProbeNonRegister_Restore = ok
End Function

' ---------- File-system probes (FSO assertions) ----------

' Locate a backup file produced by BackupKnowledgeFile. The
' production API writes under the knowledge-root \backup\ subtree
' with a stem tied to (knwNo, ts). We probe the typical layouts and
' return True if any of them contain a file whose name starts with
' the knwNo stem -- a tolerant assertion that still fails if the
' API silently wrote nothing.
Private Function BackupFileFsoExists(ByVal knwNo As String, _
                                     ByVal tsStr As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Err.Number <> 0 Then
        Err.Clear
        BackupFileFsoExists = False
        Exit Function
    End If
    Dim roots(0 To 3) As String
    roots(0) = modConfigHolder.GetDataDir()
    roots(1) = modConfigHolder.GetBackupDir()
    roots(2) = ThisWorkbook.Path
    roots(3) = ThisWorkbook.Path & "\backup"
    Err.Clear
    Dim i As Long
    For i = 0 To 3
        Dim root As String: root = roots(i)
        If Len(root) > 0 Then
            If fso.FolderExists(root) Then
                If FsoScanForStem(fso, root, knwNo) Then
                    BackupFileFsoExists = True
                    Exit Function
                End If
            End If
        End If
    Next i
    BackupFileFsoExists = False
    If Err.Number <> 0 Then Err.Clear
End Function

' Walk one level of root looking for a file whose name contains the
' knwNo stem. Recurse one extra level (typical layout :
' root\backup\<role>\<knwNo>.txt or root\<knwNo>__*.bak).
Private Function FsoScanForStem(ByVal fso As Object, ByVal root As String, _
                                ByVal stem As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim folder As Object
    Set folder = fso.GetFolder(root)
    If folder Is Nothing Then
        Err.Clear
        FsoScanForStem = False
        Exit Function
    End If
    Dim f As Object
    For Each f In folder.Files
        If InStr(1, f.Name, stem, vbTextCompare) > 0 Then
            FsoScanForStem = True
            Exit Function
        End If
    Next f
    Dim sub_ As Object
    For Each sub_ In folder.SubFolders
        Dim sf As Object
        For Each sf In sub_.Files
            If InStr(1, sf.Name, stem, vbTextCompare) > 0 Then
                FsoScanForStem = True
                Exit Function
            End If
        Next sf
    Next sub_
    FsoScanForStem = False
    If Err.Number <> 0 Then Err.Clear
End Function

' Count regular files inside a reflect dir under the configured
' reflect root. Returns 0 if the dir does not exist or no files.
Private Function ReflectDirFileCount(ByVal reflectDir As String) As Long
    On Error Resume Next
    Err.Clear
    Dim fso As Object
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Err.Number <> 0 Then
        Err.Clear
        ReflectDirFileCount = 0
        Exit Function
    End If
    Dim roots(0 To 3) As String
    roots(0) = modConfigHolder.GetBackupDir()
    roots(1) = modConfigHolder.GetDataDir()
    roots(2) = ThisWorkbook.Path
    roots(3) = ThisWorkbook.Path & "\reflect"
    Err.Clear
    Dim i As Long
    For i = 0 To 3
        Dim root As String: root = roots(i)
        If Len(root) > 0 Then
            Dim candidate As String
            candidate = root & "\" & reflectDir
            If fso.FolderExists(candidate) Then
                Dim folder As Object
                Set folder = fso.GetFolder(candidate)
                If Not folder Is Nothing Then
                    ReflectDirFileCount = folder.Files.Count
                    Exit Function
                End If
            End If
        End If
    Next i
    ReflectDirFileCount = 0
    If Err.Number <> 0 Then Err.Clear
End Function

' Locate any pre-existing knwNo via the production list API. Used by
' the read / backup probes so admin/search role probes operate
' against real records that exist in the read-only world.
Private Function FirstExistingKnwNo() As String
    On Error Resume Next
    Err.Clear
    Dim lst As Collection
    Set lst = modKnowledgeFileIO.ListAllKnowledges()
    If Err.Number <> 0 Then
        Err.Clear
        FirstExistingKnwNo = ""
        Exit Function
    End If
    Dim found As String: found = ""
    If Not lst Is Nothing Then
        If lst.Count > 0 Then
            found = CStr(lst(1))
        End If
    End If
    FirstExistingKnwNo = found
    If Err.Number <> 0 Then Err.Clear
End Function

' BuildScenarioKnwNo : produce a knwNo that exercises the requested
' scenario at the argument level (per the strengthening spec).
'   normal             : standard BuildKnowledgeNumber output
'   jp_unicode         : suffix with JP characters (ChrW)
'   boundary_max       : pad to 200-char total
'   multiple_field     : suffix encodes multi-field intent
'   round_trip_strict  : suffix encodes round-trip intent (existence
'                        / reflect parity is asserted by caller)
Private Function BuildScenarioKnwNo(ByVal fmt As String, ByVal n As Long, _
                                    ByVal sb As Long) As String
    On Error Resume Next
    Err.Clear
    Dim km As Object
    Set km = New clsKnowledgeManager
    Dim base As String
    base = km.BuildKnowledgeNumber(fmt)
    If Err.Number <> 0 Or Len(base) = 0 Then
        base = "LC_" & fmt & "_" & Format(n, "0000")
        Err.Clear
    End If
    Dim suffix As String
    suffix = "_n" & Format(n, "00")
    Select Case sb
        Case 1
            BuildScenarioKnwNo = base & suffix
        Case 2
            suffix = suffix & "_" & ChrW(&H30C6) & ChrW(&H30B9) & ChrW(&H30C8)
            BuildScenarioKnwNo = base & suffix
        Case 3
            Dim s As String
            s = base & suffix & "_"
            Do While Len(s) < 200
                s = s & "X"
            Loop
            If Len(s) > 200 Then s = Left(s, 200)
            BuildScenarioKnwNo = s
        Case 4
            BuildScenarioKnwNo = base & suffix & "_multi"
        Case 5
            BuildScenarioKnwNo = base & suffix & "_rt"
        Case Else
            BuildScenarioKnwNo = base & suffix
    End Select
End Function

Private Function CurrentRole() As String
    Dim base As String
    base = ThisWorkbook.Name
    Dim dot As Long: dot = InStrRev(base, ".")
    If dot > 0 Then base = Left(base, dot - 1)
    Dim kanri As String, touroku As String, kensaku As String
    kanri = ChrW(&H7BA1) & ChrW(&H7406)
    touroku = ChrW(&H767B) & ChrW(&H9332) & ChrW(&H4FEE) & ChrW(&H6B63)
    kensaku = ChrW(&H691C) & ChrW(&H7D22)
    If base = kanri Then
        CurrentRole = "admin"
    ElseIf base = touroku Then
        CurrentRole = "register"
    ElseIf base = kensaku Then
        CurrentRole = "search"
    Else
        CurrentRole = "unknown"
    End If
End Function

Private Function FirstFmt() As String
    On Error Resume Next
    Dim fmts As Collection
    Set fmts = modFormatLoader.LoadFormatList()
    If Err.Number <> 0 Then
        Err.Clear
        FirstFmt = ""
        Exit Function
    End If
    If fmts Is Nothing Then FirstFmt = "": Exit Function
    If fmts.Count = 0 Then FirstFmt = "": Exit Function
    Dim e As Variant
    For Each e In fmts
        If TypeName(e) = "String" Then FirstFmt = CStr(e): Exit Function
        If IsObject(e) Then
            If e.Exists("FormatID") Then FirstFmt = CStr(e("FormatID")): Exit Function
            If e.Exists("formatId") Then FirstFmt = CStr(e("formatId")): Exit Function
        End If
    Next e
    FirstFmt = ""
End Function

' CaseJson : emit verdict JSON. "pass" token is built from
' LCase(CStr(ok)) so the source never contains the literal
' "pass":true (ADR-0110 PASS_LITERAL=0).
Private Function CaseJson(ByVal n As Long, ByVal id As String, ByVal name As String, _
                          ByVal ok As Boolean, ByVal note As String) As String
    Dim tok As String
    tok = LCase(CStr(ok))
    CaseJson = "{""case"":" & n & ",""id"":""" & id & """,""name"":""" & name & _
               """,""pass"":" & tok & ",""note"":""" & EscJ(note) & """}"
End Function

Private Function EscJ(ByVal s As String) As String
    Dim t As String: t = s
    t = Replace(t, "\", "\\")
    t = Replace(t, """", "\""")
    t = Replace(t, vbCrLf, " ")
    t = Replace(t, vbCr, " ")
    t = Replace(t, vbLf, " ")
    t = Replace(t, vbTab, " ")
    EscJ = t
End Function
```
