## 保存方法

下のコードをメモ帳に貼り付け、**[名前を付けて保存]** で次のように保存してください。

- 場所: `C:\KnowledgeMgr\installer\vba_modules\common\\`
- ファイル名: `modE2E_MultiRole.bas`
- ファイルの種類: **すべてのファイル**
- 文字コード: **ANSI**（Shift-JIS）

> メモ帳の文字コードを **ANSI** にしないと、VBA の日本語が文字化けして動かなくなります。
> UTF-8 で保存すると VBA Import 時に日本語が文字化けして動かなくなります。
> 改行コードは CRLF（Windows 標準）のままで OK です。

## ソースコード

```vb
Attribute VB_Name = "modE2E_MultiRole"
' ================================================================
' modE2E_MultiRole  (v2.3 E2E test, multi-role isolation)
' Purpose:
'   15 case (5 category x 3 role) that verify production-code role
'   gates / propagation by calling REAL APIs (modKnowledgeFileIO,
'   modFormatLoader, clsKnowledgeManager). No "pass":true literal.
'   Per ADR-0092: FAIL when assertion fails, "skipped-by-design"
'   when current xlsm role does not match the case target role.
'   Per ADR-0094 D1: this module is generated by Python (CP932/CRLF).
'   Per ADR-0090: no hard-coded role/format/cell literal; role detected
'   via ThisWorkbook.Name compared to ChrW-derived names.
' ================================================================
Option Explicit

' ---------- Public entry points (signature preserved) -----------

Public Function Run_MultiRole_Cases(ByVal role As String) As String
    Dim sb As String, sep As String
    sb = "{""role"":""multirole-" & role & """,""cases"":["
    sep = ""
    Dim curRole As String
    curRole = CurrentRole()
    Dim i As Long
    For i = 1 To 15
        sb = sb & sep & RunMr(i, curRole)
        sep = ","
    Next i
    sb = sb & "]}"
    Run_MultiRole_Cases = sb
End Function

Public Sub Run_MultiRole_Cases_Out(ByVal outPath As String)
    Dim role As String, s As String, fh As Integer
    role = Replace(ThisWorkbook.Name, ".xlsm", "")
    s = Run_MultiRole_Cases(role)
    fh = FreeFile
    Open outPath For Output As #fh
    Print #fh, s
    Close #fh
End Sub

' ---------- Case dispatcher ----------

Private Function RunMr(ByVal n As Long, ByVal curRole As String) As String
    Dim id As String, name As String, note As String, cat As String, tgtRole As String
    Dim ok As Boolean
    ' Default: assertion has not yet passed. Each Assert* helper must set
    ' ok = True before returning. Unhandled paths therefore emit the literal
    ' FAIL JSON below ("pass":false).
    ok = False

    id = "mr-" & Format(n, "00")
    Dim grp As Long: grp = ((n - 1) \ 3) + 1
    Select Case grp
        Case 1: cat = "role_isolation"
        Case 2: cat = "format_propagation"
        Case 3: cat = "knowledge_propagation"
        Case 4: cat = "id_consistency"
        Case 5: cat = "button_visibility"
    End Select
    Dim subN As Long: subN = ((n - 1) Mod 3) + 1
    Select Case subN
        Case 1: tgtRole = "admin"
        Case 2: tgtRole = "register"
        Case 3: tgtRole = "search"
    End Select
    name = cat & "_" & tgtRole

    If curRole <> tgtRole Then
        note = "skipped-by-design curRole=" & curRole & " tgtRole=" & tgtRole
        RunMr = "{""case"":" & n & ",""id"":""" & id & """,""name"":""" & name & _
                """,""ok"":""skipped-by-design"",""note"":""" & EscJ(note) & """}"
        Exit Function
    End If

    Select Case grp
        Case 1: ok = AssertRoleIsolation(tgtRole, note)
        Case 2: ok = AssertFormatPropagation(tgtRole, note)
        Case 3: ok = AssertKnowledgePropagation(tgtRole, note)
        Case 4: ok = AssertIdConsistency(tgtRole, note)
        Case 5: ok = AssertButtonVisibility(tgtRole, note)
    End Select

    If ok Then
        RunMr = "{""case"":" & n & ",""id"":""" & id & """,""name"":""" & name & _
                """,""pass"":true,""note"":""" & EscJ(note) & """}"
    Else
        RunMr = "{""case"":" & n & ",""id"":""" & id & """,""name"":""" & name & _
                """,""pass"":false,""note"":""" & EscJ(note) & """}"
    End If
End Function

' ================================================================
' Category 1: role_isolation
'   admin: SaveKnowledge should be rejected (rc=3)
'   register: SaveFormat should be rejected (rc<>0)
'   search: both should be rejected
' ================================================================
Private Function AssertRoleIsolation(ByVal tgt As String, ByRef note As String) As Boolean
    On Error Resume Next
    Dim rcK As Long, rcF As Long
    Dim d As Object
    Set d = CreateObject("Scripting.Dictionary")
    d("formatId") = "ROLE_ISO_PROBE"
    d("probe") = "role_isolation_" & tgt
    Err.Clear
    Select Case tgt
        Case "admin"
            rcK = modKnowledgeFileIO.SaveKnowledge("ROLE_ISO_PROBE-0001", d, 0)
            note = "admin SaveKnowledge rc=" & rcK & " expect=3"
            AssertRoleIsolation = (rcK = 3)
        Case "register"
            Dim secs As Collection
            Set secs = New Collection
            rcF = modFormatLoader.SaveFormat("ROLE_ISO_PROBE", secs)
            note = "register SaveFormat rc=" & rcF & " expect_nonzero"
            AssertRoleIsolation = (rcF <> 0)
        Case "search"
            rcK = modKnowledgeFileIO.SaveKnowledge("ROLE_ISO_PROBE-0001", d, 0)
            Dim secs2 As Collection
            Set secs2 = New Collection
            rcF = modFormatLoader.SaveFormat("ROLE_ISO_PROBE", secs2)
            note = "search SaveKnowledge rc=" & rcK & " SaveFormat rc=" & rcF & " expect both gated"
            AssertRoleIsolation = ((rcK = 3) And (rcF <> 0))
        Case Else
            note = "unknown tgt=" & tgt
            AssertRoleIsolation = False
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        AssertRoleIsolation = False
    End If
End Function

' ================================================================
' Category 2: format_propagation
'   admin: SaveFormat round-trip
'   register: LoadFormat read-only success
'   search: LoadFormat read-only success
' ================================================================
Private Function AssertFormatPropagation(ByVal tgt As String, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim fmt As String
    fmt = FirstFmt()
    If Len(fmt) = 0 Then
        note = "no_format_in_system"
        AssertFormatPropagation = False
        Exit Function
    End If
    Select Case tgt
        Case "admin"
            Dim secs1 As Collection
            Set secs1 = modFormatLoader.LoadFormat(fmt)
            If secs1 Is Nothing Then
                note = "load_pre_nothing fmt=" & fmt
                AssertFormatPropagation = False
                Exit Function
            End If
            Dim cnt1 As Long: cnt1 = secs1.Count
            Dim rc As Long
            rc = modFormatLoader.SaveFormat(fmt, secs1)
            Dim secs2 As Collection
            Set secs2 = modFormatLoader.LoadFormat(fmt)
            Dim cnt2 As Long: cnt2 = 0
            If Not secs2 Is Nothing Then cnt2 = secs2.Count
            note = "admin SaveFormat rc=" & rc & " pre_cnt=" & cnt1 & " post_cnt=" & cnt2
            AssertFormatPropagation = ((rc = 0) And (cnt1 > 0) And (cnt1 = cnt2))
        Case "register", "search"
            Dim secsR As Collection
            Set secsR = modFormatLoader.LoadFormat(fmt)
            Dim cntR As Long: cntR = 0
            If Not secsR Is Nothing Then cntR = secsR.Count
            note = tgt & " LoadFormat fmt=" & fmt & " sec_count=" & cntR
            AssertFormatPropagation = (cntR > 0)
        Case Else
            note = "unknown tgt=" & tgt
            AssertFormatPropagation = False
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        AssertFormatPropagation = False
    End If
End Function

' ================================================================
' Category 3: knowledge_propagation
'   register: SaveKnowledge -> LoadKnowledge round-trip
'   admin/search: SaveKnowledge rejected (rc=3)
' ================================================================
Private Function AssertKnowledgePropagation(ByVal tgt As String, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim fmt As String
    fmt = FirstFmt()
    If Len(fmt) = 0 Then
        note = "no_format_in_system"
        AssertKnowledgePropagation = False
        Exit Function
    End If
    Dim probeNo As String
    probeNo = fmt & "-9991"
    Dim d As Object
    Set d = CreateObject("Scripting.Dictionary")
    d("formatId") = fmt
    d("propagation_probe") = "kp_" & tgt & "_" & Format(Now, "hhnnss")
    Select Case tgt
        Case "register"
            Dim rc As Long
            rc = modKnowledgeFileIO.SaveKnowledge(probeNo, d, 0)
            If rc <> 0 Then
                note = "register SaveKnowledge rc=" & rc & " expect=0"
                AssertKnowledgePropagation = False
                Exit Function
            End If
            Dim got As Object
            Set got = modKnowledgeFileIO.LoadKnowledge(probeNo)
            Dim okLoaded As Boolean
            okLoaded = False
            If Not got Is Nothing Then
                If got.Exists("propagation_probe") Then okLoaded = True
            End If
            On Error Resume Next
            modKnowledgeFileIO.DeleteKnowledge probeNo
            On Error Resume Next
            note = "register round_trip save=0 load_ok=" & okLoaded
            AssertKnowledgePropagation = okLoaded
        Case "admin", "search"
            Dim rcG As Long
            rcG = modKnowledgeFileIO.SaveKnowledge(probeNo, d, 0)
            note = tgt & " SaveKnowledge rc=" & rcG & " expect=3 (gated)"
            AssertKnowledgePropagation = (rcG = 3)
        Case Else
            note = "unknown tgt=" & tgt
            AssertKnowledgePropagation = False
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        AssertKnowledgePropagation = False
    End If
End Function

' ================================================================
' Category 4: id_consistency
'   register: BuildKnowledgeNumber twice, persist id1, expect seq2-seq1=1
'   admin/search: BuildKnowledgeNumber non-empty
' ================================================================
Private Function AssertIdConsistency(ByVal tgt As String, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim fmt As String
    fmt = FirstFmt()
    If Len(fmt) = 0 Then
        note = "no_format_in_system"
        AssertIdConsistency = False
        Exit Function
    End If
    Dim mgr As Object
    Set mgr = New clsKnowledgeManager
    mgr.Init Nothing, Nothing, ""
    Select Case tgt
        Case "register"
            Dim id1 As String, id2 As String
            id1 = mgr.BuildKnowledgeNumber(fmt)
            Dim d As Object
            Set d = CreateObject("Scripting.Dictionary")
            d("formatId") = fmt
            d("idconsistency_probe") = "ic_" & Format(Now, "hhnnss")
            Dim rcSave As Long
            rcSave = modKnowledgeFileIO.SaveKnowledge(id1, d, 0)
            id2 = mgr.BuildKnowledgeNumber(fmt)
            Dim seq1 As Long, seq2 As Long
            seq1 = ParseSeqTail(id1)
            seq2 = ParseSeqTail(id2)
            On Error Resume Next
            modKnowledgeFileIO.DeleteKnowledge id1
            On Error Resume Next
            note = "register id1=" & id1 & " id2=" & id2 & " save_rc=" & rcSave & _
                   " seq1=" & seq1 & " seq2=" & seq2 & " expect_diff=1"
            AssertIdConsistency = ((rcSave = 0) And (seq2 - seq1 = 1) And (seq1 > 0))
        Case "admin", "search"
            Dim idR As String
            idR = mgr.BuildKnowledgeNumber(fmt)
            note = tgt & " BuildKnowledgeNumber=" & idR & " expect_nonempty"
            AssertIdConsistency = (Len(idR) > 0)
        Case Else
            note = "unknown tgt=" & tgt
            AssertIdConsistency = False
    End Select
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        AssertIdConsistency = False
    End If
End Function

' ================================================================
' Category 5: button_visibility
'   Count Shapes across all worksheets; PASS when at least 1 exists.
'   Also reports Btn_ prefix count for diagnostic.
' ================================================================
Private Function AssertButtonVisibility(ByVal tgt As String, ByRef note As String) As Boolean
    On Error Resume Next
    Err.Clear
    Dim totalShapes As Long
    Dim btnShapes As Long
    Dim sheetCount As Long
    Dim ws As Worksheet
    For Each ws In ThisWorkbook.Worksheets
        sheetCount = sheetCount + 1
        Dim s As Shape
        For Each s In ws.Shapes
            totalShapes = totalShapes + 1
            If Left(s.Name, 4) = "Btn_" Then btnShapes = btnShapes + 1
        Next s
    Next ws
    note = tgt & " sheets=" & sheetCount & " total_shapes=" & totalShapes & _
           " btn_shapes=" & btnShapes & " expect>0"
    AssertButtonVisibility = (totalShapes > 0)
    If Err.Number <> 0 Then
        note = note & " err=" & Err.Description
        Err.Clear
        AssertButtonVisibility = False
    End If
End Function

' ================================================================
' Helpers
' ================================================================

Private Function CurrentRole() As String
    Dim base As String
    base = ThisWorkbook.Name
    Dim dot As Long: dot = InStrRev(base, ".")
    If dot > 0 Then base = Left(base, dot - 1)
    Dim kanri As String, touroku As String, kensaku As String
    kanri = ChrW(&H7BA1) & ChrW(&H7406)
    touroku = ChrW(&H767B) & ChrW(&H9332)
    kensaku = ChrW(&H691C) & ChrW(&H7D22)
    If base = kanri Then
        CurrentRole = "admin"
    ElseIf base = touroku Then
        CurrentRole = "register"
    ElseIf base = kensaku Then
        CurrentRole = "search"
    Else
        CurrentRole = "unknown"
    End If
End Function

Private Function FirstFmt() As String
    On Error Resume Next
    Dim fmts As Collection
    Set fmts = modFormatLoader.LoadFormatList()
    If Err.Number <> 0 Then
        Err.Clear
        FirstFmt = ""
        Exit Function
    End If
    If fmts Is Nothing Then FirstFmt = "": Exit Function
    If fmts.Count = 0 Then FirstFmt = "": Exit Function
    Dim e As Variant
    For Each e In fmts
        If TypeName(e) = "String" Then FirstFmt = CStr(e): Exit Function
        If IsObject(e) Then
            If e.Exists("FormatID") Then FirstFmt = CStr(e("FormatID")): Exit Function
            If e.Exists("formatId") Then FirstFmt = CStr(e("formatId")): Exit Function
        End If
    Next e
    FirstFmt = ""
End Function

Private Function ParseSeqTail(ByVal knwNo As String) As Long
    Dim dash As Long: dash = InStrRev(knwNo, "-")
    If dash <= 0 Then ParseSeqTail = 0: Exit Function
    Dim tail As String
    tail = Mid(knwNo, dash + 1)
    On Error Resume Next
    ParseSeqTail = CLng(tail)
    If Err.Number <> 0 Then ParseSeqTail = 0: Err.Clear
End Function

Private Function EscJ(ByVal s As String) As String
    Dim t As String: t = s
    t = Replace(t, "\", "\\")
    t = Replace(t, """", "\""")
    t = Replace(t, vbCrLf, " ")
    t = Replace(t, vbCr, " ")
    t = Replace(t, vbLf, " ")
    t = Replace(t, vbTab, " ")
    EscJ = t
End Function
```
