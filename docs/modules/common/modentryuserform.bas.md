## 保存方法

下のコードをメモ帳に貼り付け、**[名前を付けて保存]** で次のように保存してください。

- 場所: `C:\KnowledgeMgr\installer\vba_modules\common\\`
- ファイル名: `modEntryUserForm.bas`
- ファイルの種類: **すべてのファイル**
- 文字コード: **ANSI**（Shift-JIS）

> メモ帳の文字コードを **ANSI** にしないと、VBA の日本語が文字化けして動かなくなります。
> UTF-8 で保存すると VBA Import 時に日本語が文字化けして動かなくなります。
> 改行コードは CRLF（Windows 標準）のままで OK です。

' ================================================================
' Public Function: TestBuildOnly_ShowForm
' Phase N-5 E2E helper. Performs the same logic as ShowForm but
' skips the actual vbModal Show + dispose so a headless test runner
' can verify the build-form path and the persist-and-return path
' deterministically. Returns the candidate knowledgeId that WOULD
' be generated for register mode (or the loaded id for edit/view).
' ================================================================
Public Function TestBuildOnly_ShowForm( _
        ByVal mode As String, _
        ByVal knowledgeId As String, _
        ByVal formatId As String, _
        ByVal fieldsDict As Object _
) As String
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1123] modEntryUserForm.TestBuildOnly_ShowForm ENTER"  ' [ADR-0100]
    On Error GoTo ErrHandler
    ' Persist directly via modUserFormCallback's internal helper logic.
    ' We bypass the actual modal show and exercise the file I/O round
    ' trip that the OnRegister / OnUpdate handlers would normally do.
    Dim sb As String
    Dim id As String
    Select Case LCase(mode)
        Case "register"
            id = modUserFormCallback.TestGenerateKnowledgeId(formatId)
        Case "edit"
            id = knowledgeId
        Case Else
            id = knowledgeId
    End Select

    sb = "###KnowledgeNo###" & vbCrLf & id & vbCrLf
    sb = sb & "###FormatID###" & vbCrLf & formatId & vbCrLf

    Dim k As Variant
    For Each k In fieldsDict.Keys
        sb = sb & "###" & CStr(k) & "###" & vbCrLf & CStr(fieldsDict(CStr(k))) & vbCrLf
    Next k

    Dim filePath As String
    filePath = modConfigHolder.GetDataDir() & id & ".txt"
    Dim adoStream As Object
    Set adoStream = CreateObject("ADODB.Stream")
    adoStream.Type = 2
    adoStream.Charset = "Shift_JIS"
    adoStream.Open
    adoStream.WriteText sb
    adoStream.SaveToFile filePath, 2
    adoStream.Close

    TestBuildOnly_ShowForm = id
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1124] modEntryUserForm.TestBuildOnly_ShowForm EXIT-OK"  ' [ADR-0100]
    Exit Function
ErrHandler:
    If modCommon.gDebugLevel >= DEBUG_LEVEL_ERROR Then Debug.Print "[D-1125] modEntryUserForm.TestBuildOnly_ShowForm EXIT-ERR " & "errNum=" & Err.Number & " desc=" & Err.Description  ' [ADR-0100]
    Debug.Print "[ERR] TestBuildOnly_ShowForm: " & Err.Number & " " & Err.Description
    TestBuildOnly_ShowForm = ""
End Function

' Internal helper that lets the test entry pin a specific formatId. Reuses
' the production ShowForm dispatch path; only the seed formatId differs.
Private Function TestShowFormWithFmt( _
        ByVal xlsmName As String, _
        ByVal mode As String, _
        ByVal knowledgeId As String, _
        ByVal formatId As String _
) As String
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1129] modEntryUserForm.TestShowFormWithFmt ENTER"  ' [ADR-0100]
    Dim r As clsUserFormRenderer
    Set r = New clsUserFormRenderer
    ' Workaround: ShowForm normally resolves formatId from the loaded
    ' knowledge file. For register mode we need to seed it externally.
    ' Use the Public Property GetFormatId style; but the renderer keeps
    ' m_formatId private. Easiest: write a wrapper that calls the same
    ' BuildAndShow flow via a public stub on the renderer if available;
    ' otherwise fall back to ShowForm() (which resolves from the file
    ' when knowledgeId is given - works for edit/view but not register).
    r.TestSeedFormatId formatId
    TestShowFormWithFmt = r.ShowForm(xlsmName, mode, knowledgeId)
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1130] modEntryUserForm.TestShowFormWithFmt EXIT-OK"  ' [ADR-0100]
End Function

' Phase R-3-gamma-2 E2E (case a): 実 persistence path を Unicode 安全に駆動する

'   xlsmName: 役割名 / mode: register|edit / knowledgeId / formatId
'   action  : register|update|delete / valuesFile: UTF-8 値ファイル(空可)
' ================================================================
Public Function TestE2E_DriveForm(ByVal xlsmName As String, ByVal mode As String, _
        ByVal knowledgeId As String, ByVal formatId As String, _
        ByVal action As String, ByVal valuesFile As String) As String
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1146] modEntryUserForm.TestE2E_DriveForm ENTER"  ' [ADR-0100]
    On Error GoTo EH
    Dim r As clsUserFormRenderer
    Set r = New clsUserFormRenderer
    r.TestSeedFormatId formatId
    modUserFormCallback.SetRenderer r
    Dim res As String
    res = r.ShowFormModeless(xlsmName, mode, knowledgeId, formatId)
    Dim frm As Object, uf As Object
    For Each uf In VBA.UserForms
        If Left(uf.Name, 7) = "frmDyn_" Then Set frm = uf
    Next uf
    If frm Is Nothing Then
        TestE2E_DriveForm = "ERR:no_form res=" & res
        If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1147] modEntryUserForm.TestE2E_DriveForm EXIT-OK"  ' [ADR-0100]
        Exit Function
    End If
    If Len(valuesFile) > 0 Then E2E_ApplyValues frm, valuesFile
    Select Case LCase(action)
        Case "register": modUserFormCallback.OnRegister
        Case "update":   modUserFormCallback.OnUpdate
        Case "delete":   modUserFormCallback.OnDelete
        Case Else        ' "none" = 値設定/保存せず(表示確認のみ)
    End Select
    r.CloseModelessForm
    TestE2E_DriveForm = "OK:" & res
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1148] modEntryUserForm.TestE2E_DriveForm EXIT-OK"  ' [ADR-0100]
    Exit Function
EH:
    TestE2E_DriveForm = "ERR:" & Err.Number & ":" & Err.Description
End Function

Private Sub E2E_ApplyValues(ByVal frm As Object, ByVal valuesFile As String)
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1149] modEntryUserForm.E2E_ApplyValues ENTER"  ' [ADR-0100]
    On Error Resume Next
    Dim st As Object
    Set st = CreateObject("ADODB.Stream")
    st.Type = 2: st.Charset = "UTF-8": st.Open
    st.LoadFromFile valuesFile
    Dim all As String
    all = st.ReadText
    st.Close
    all = Replace(all, vbCrLf, vbLf)
    all = Replace(all, vbCr, vbLf)
    Dim lines() As String, i As Long
    lines = Split(all, vbLf)
    For i = LBound(lines) To UBound(lines)
        Dim ln As String
        ln = lines(i)
        Dim eq As Long
        eq = InStr(ln, "=")
        If eq > 1 Then
            Dim cn As String, vv As String
            cn = Left(ln, eq - 1)
            vv = Mid(ln, eq + 1)
            vv = Replace(vv, "\n", vbCrLf)   ' unescape newline marker
            Dim ctl As Object
            Set ctl = E2E_FindCtl(frm, cn)
            If Not ctl Is Nothing Then
                If TypeName(ctl) = "ComboBox" Then
                    ctl.value = vv
                Else
                    ctl.Text = vv
                End If
                ctl.Tag = ""   ' clear placeholder tag so value is not treated as empty
            End If
        End If
    Next i

' ADR-0090 (2026-06-01): Resolve "knowledge id" cell via SSOT lookup.
' Active sheet name -> screen ID (modSheetMap.SheetToScreenId) -> for each
' role dir (admin/search/register, CJK names per ADR-0089) try
' LoadUiDefinition; inspect [INPUT] sections for inputDataKey in
' {"kid","knowledgeId","knwId","knowledge_id"} and use that Cell.
' Falls back to legacy B3 hard-code so legacy installs that lack the
' new inputDataKey still work.
Private Function ReadKnowledgeIdFromActiveSheet() As String
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1158] modEntryUserForm.ReadKnowledgeIdFromActiveSheet ENTER"  ' [ADR-0100]
    On Error GoTo Fallback
    Dim ws As Object
    Set ws = ActiveSheet
    If ws Is Nothing Then GoTo Fallback
    Dim cellAddr As String
    cellAddr = ResolveKidCellFromActiveSheet(ws)
    If Len(cellAddr) = 0 Then cellAddr = "B3"
    Dim v As Variant

    If Not IsEmpty(v) Then ReadKnowledgeIdFromActiveSheet = CStr(v)
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1159] modEntryUserForm.ReadKnowledgeIdFromActiveSheet EXIT-OK"  ' [ADR-0100]
    Exit Function
Fallback:
    On Error Resume Next
    Dim v2 As Variant
    v2 = ActiveSheet.Range("B3").Value
    If Not IsEmpty(v2) Then ReadKnowledgeIdFromActiveSheet = CStr(v2)
    On Error GoTo 0
End Function

' ADR-0090 helper: SSOT-driven kid cell address resolution
' across 3 candidate role dirs (admin/search/register).
Private Function ResolveKidCellFromActiveSheet(ByVal ws As Object) As String
    If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1160] modEntryUserForm.ResolveKidCellFromActiveSheet ENTER"  ' [ADR-0100]
    On Error GoTo Fallback
    Dim screenId As String
    screenId = modSheetMap.SheetToScreenId(ws.Name)
    If Len(screenId) = 0 Then GoTo Fallback
    Dim roles(0 To 2) As String
    roles(0) = RoleAdminLocal()
    roles(1) = RoleSearchLocal()
    roles(2) = RoleRegisterLocal()
    Dim r As Long
    For r = 0 To 2
        Dim sections As Collection
        Set sections = modUILoader.LoadUiDefinition(roles(r), screenId)
        If Not sections Is Nothing Then
            If sections.Count > 0 Then
                Dim i As Long
                For i = 1 To sections.Count
                    Dim sec As ClsStanzaSection
                    Set sec = sections.Item(i)
                    If sec.SectionName = "INPUT" Then
                        Dim key As String
                        key = sec.GetValue("inputDataKey")
                        If key = "kid" Or key = "knowledgeId" Or _
                           key = "knwId" Or key = "knowledge_id" Then
                            Dim cellExpr As String
                            cellExpr = sec.GetValue("Cell")
                            If Len(cellExpr) > 0 Then
                                Dim p As Long
                                p = InStr(1, cellExpr, ":")
                                If p > 0 Then
                                    ResolveKidCellFromActiveSheet = Left(cellExpr, p - 1)
                                Else
                                    ResolveKidCellFromActiveSheet = cellExpr
                                End If
                                If modCommon.gDebugLevel >= DEBUG_LEVEL_TRACE Then Debug.Print "[D-1161] modEntryUserForm.ResolveKidCellFromActiveSheet EXIT-OK"  ' [ADR-0100]
                                Exit Function
                            End If
                        End If
                    End If
                Next i
            End If
        End If
    Next r
Fallback:
    ResolveKidCellFromActiveSheet = ""
End Function
